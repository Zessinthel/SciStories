\usepackage{titlesec}

% Capítulo: formato en línea, sin "Capítulo", solo número y título
\titleformat{\chapter}
  {\normalfont\Huge\bfseries\color{CtpPink}}
  {\thechapter\space}{0pt}{}

% Espaciado: menos espacio antes y después
\titlespacing*{\chapter}{0pt}{0pt}{20pt}

% Sección
\titleformat{\section}
  {\normalfont\Large\bfseries\color{CtpLavender}}
  {\thesection}{1em}{}

% Subsección
\titleformat{\subsection}
  {\normalfont\large\bfseries\color{CtpSky}}
  {\thesubsection}{1em}{}

% Subsubsección
\titleformat{\subsubsection}
  {\normalfont\normalsize\bfseries\color{CtpGreen}}
  {\thesubsubsection}{1em}{}

% Ajuste de espaciado de secciones (opcional)
\titlespacing*{\section}{0pt}{3.5ex plus 1ex minus .2ex}{2.3ex plus .2ex}
\titlespacing*{\subsection}{0pt}{3.25ex plus 1ex minus .2ex}{1.5ex plus .2ex}
\titlespacing*{\subsubsection}{0pt}{3.25ex plus 1ex minus .2ex}{1.5ex plus .2ex}
